sol-accordion { @include flex; flex-flow: column; > sol-accordion-item { display: grid; grid-template-columns: 1fr; grid-template-rows: auto 1fr; cursor: pointer; > .header { @include flex; justify-content: space-between; > svg { transform: rotate(0turn); transition: transform 125ms cubic-bezier(0.455, 0.03, 0.515, 0.955); } } &.active { > .header { > svg { transform: rotate(0.5turn); } } > .content { grid-template-rows: 1fr; transition: grid-template-rows 175ms cubic-bezier(0.455, 0.03, 0.515, 0.955); } } > .content { display: grid; grid-template-rows: 0fr; transition: grid-template-rows 225ms cubic-bezier(0.455, 0.03, 0.515, 0.955); padding: rem(get($spacings, small)) rem(get($spacings, one)) rem(get($spacings, one)); > .content-inner { overflow: hidden; } } } } const sol_accordion = () => { let accs = document.querySelectorAll(`sol-accordion > sol-accordion-item`); for (const a of accs) { a.addEventListener("click", () => { let others = a.parentElement.querySelector(`.active`); if (others !== a) { others?.classList.remove(`active`); } a.classList.toggle(`active`); }); } }; export { sol_accordion };
